<!DOCTYPE html>

<html lang="en">

<head>

  <meta charset="UTF-8">

  <meta name="viewport" content="width=device-width, initial-scale=1.0">

  <title>Jira Story Parser</title>

  <link href="https://cdn.jsdelivr.net/npm/tailwindcss@2.2.16/dist/tailwind.min.css" rel="stylesheet">

</head>

<body>

  <div id="root"></div>


  <script src="https://cdnjs.cloudflare.com/ajax/libs/react/17.0.2/umd/react.production.min.js"></script>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/react-dom/17.0.2/umd/react-dom.production.min.js"></script>

  <script>

    const { useState } = React;


    const JiraStoryParser = () => {

      const [jiraUrl, setJiraUrl] = useState('');

      const [username, setUsername] = useState('');

      const [password, setPassword] = useState('');

      const [stories, setStories] = useState([]);

      const [testCases, setTestCases] = useState([]);


      const parseJiraStories = async () => {

        try {

          const response = await fetch(jiraUrl, {

            headers: {

              'Authorization': `Basic ${btoa(`${username}:${password}`)}`

            }

          });


          const data = await response.json();

          const parsedStories = await parseStories(data.issues);

          setStories(parsedStories);


          const generatedTestCases = generateTestCases(parsedStories);

          setTestCases(generatedTestCases);

        } catch (error) {

          console.error('Error parsing Jira stories:', error);

        }

      };


      const parseStories = async (issuesArray) => {

        const stories = [];


        for (const issueJson of issuesArray) {

          const story = new JiraStory();

          story.setId(issueJson.id);

          story.setSummary(issueJson.fields.summary);

          story.setDescription(issueJson.fields.description);


          const linkedStories = await fetchLinkedStories(issueJson.fields.issuelinks);

          story.setLinkedStories(linkedStories);


          stories.push(story);

        }


        return stories;

      };


      const fetchLinkedStories = async (linkedIssuesArray) => {

        const linkedStories = [];


        for (const linkedIssueJson of linkedIssuesArray) {

          if (linkedIssueJson.outwardIssue) {

            const linkedStory = new JiraStory();

            linkedStory.setId(linkedIssueJson.outwardIssue.id);

            linkedStory.setSummary(linkedIssueJson.outwardIssue.fields.summary);

            linkedStory.setDescription(linkedIssueJson.outwardIssue.fields.description);

            linkedStories.push(linkedStory);

          }

        }


        return linkedStories;

      };


      const generateTestCases = (stories) => {

        const testCases = [];


        for (const story of stories) {

          const testCase = new TestCase();

          testCase.setTitle(story.getSummary());

          testCase.setDescription(story.getDescription());

          testCase.setLinkedStories(story.getLinkedStories());

          testCases.push(testCase);

        }


        return testCases;

      };


      return (

        <div className="container mx-auto py-8">

          <h1 className="text-3xl font-bold mb-4">Jira Story Parser</h1>

          <div className="mb-4">

            <label htmlFor="jiraUrl" className="block font-medium mb-2">Jira URL:</label>

            <input

              id="jiraUrl"

              type="text"

              className="border rounded px-3 py-2 w-full"

              value={jiraUrl}

              onChange={(e) => setJiraUrl(e.target.value)}

            />

          </div>

          <div className="mb-4">

            <label htmlFor="username" className="block font-medium mb-2">Username:</label>

            <input

              id="username"

              type="text"

              className="border rounded px-3 py-2 w-full"

              value={username}

              onChange={(e) => setUsername(e.target.value)}

            />

          </div>

          <div className="mb-4">

            <label htmlFor="password" className="block font-medium mb-2">Password:</label>

            <input

              id="password"

              type="password"

              className="border rounded px-3 py-2 w-full"

              value={password}

              onChange={(e) => setPassword(e.target.value)}

            />

          </div>

          <button

            className="bg-blue-500 hover:bg-blue-700 text-white font-bold py-2 px-4 rounded"

            onClick={parseJiraStories}

          >

            Parse Jira Stories

          </button>


          {stories.length > 0 && (

            <div className="mt-8">

              <h2 className="text-2xl font-bold mb-4">Jira Stories</h2>

              <ul>

                {stories.map((story) => (

                  <li key={story.id} className="border rounded p-4 mb-4">

                    <h3 className="text-xl font-bold mb-2">{story.summary}</h3>

                    <p className="mb-4">{story.description}</p>

                    {story.linkedStories.length > 0 && (

                      <div>

                        <h4 className="text-lg font-bold mb-2">Linked Stories:</h4>

                        <ul>

                          {story.linkedStories.map((linkedStory) => (

                            <li key={linkedStory.id} className="mb-2">

                              <h5 className="font-bold">{linkedStory.summary}</h5>

                              <p>{linkedStory.description}</p>

                            </li>

                          ))}

                        </ul>

                      </div>

                    )}

                  </li>

                ))}

              </ul>

            </div>

          )}


          {testCases.length > 0 && (

            <div className="mt-8">

              <h2 className="text-2xl font-bold mb-4">Test Cases</h2>

              <ul>

                {testCases.map((testCase, index) => (

                  <li key={index} className="border rounded p-4 mb-4">

                    <h3 className="text-xl font-bold mb-2">{testCase.title}</h3>

                    <p className="mb-4">{testCase.description}</p>

                    {testCase.linkedStories.length > 0 && (

                      <div>

                        <h4 className="text-lg font-bold mb-2">Linked Stories:</h4>

                        <ul>

                          {testCase.linkedStories.map((linkedStory, index) => (

                            <li key={index} className="mb-2">

                              <h5 className="font-bold">{linkedStory.summary}</h5>

                              <p>{linkedStory.description}</p>

                            </li>

                          ))}

                        </ul>

                      </div>

                    )}

                  </li>

                ))}

              </ul>

            </div>

          )}

        </div>

      );

    };


    class JiraStory {

      constructor() {

        this.id = '';

        this.summary = '';

        this.description = '';

        this.linkedStories = [];

      }


      setId(id) {

        this.id = id;

      }


      setSummary(summary) {

        this.summary = summary;

      }


      setDescription(description) {

        this.description = description;

      }


      setLinkedStories(linkedStories) {

        this.linkedStories = linkedStories;

      }


      getLinkedStories() {

        return this.linkedStories;

      }

    }


    class TestCase {

      constructor() {

        this.title = '';

        this.description = '';

        this.linkedStories = [];

      }


      setTitle(title) {

        this.title = title;

      }


      setDescription(description) {

        this.description = description;

      }


      setLinkedStories(linkedStories) {

        this.linkedStories = linkedStories;

      }

    }


    ReactDOM.render(<JiraStoryParser />, document.getElementById('root'));

  </script>

</body>

</html>
